function groupMessages(msgs) {
  const out = [];
  let cur = null;
  let lastDay = null;
  msgs.forEach(m => {
    if (m.day && m.day !== lastDay) {
      out.push({
        divider: m.day,
        msgs: []
      });
      lastDay = m.day;
      cur = null;
    }
    if (out.length === 0) out.push({
      msgs: []
    });
    const block = out[out.length - 1];
    if (block.msgs.length === 0 || block.msgs[block.msgs.length - 1].author !== m.author) {
      block.msgs.push(m);
      cur = block;
    } else {
      block.msgs.push(m);
    }
  });
  return out;
